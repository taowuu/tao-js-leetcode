```javascript
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    // 数组元素不能重复出现想到哈希表
    const obj = {}
    for(let i = 0; i < nums.length; i++) {
        // 例如遇到 2 则寻找 7 是否在缓存中
        let diff = target - nums[i]
        if(diff in obj) {
            // 找到则返回答案
            return [i, obj[diff]]
        } else {
            // 没找到则先缓存遍历过的值
            obj[nums[i]] = i 
        }
    }
};
```
