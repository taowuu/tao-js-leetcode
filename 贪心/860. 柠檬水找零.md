### 代码
```javascript
/**
 * @param {number[]} bills
 * @return {boolean}
 */
var lemonadeChange = function(bills) {
    // 五元和十元可作为零钱
    let five = 0,
        ten = 0
    for(let i = 0; i < bills.length; i++) {
        if(bills[i] === 5) {
            // 五元则增加零钱
            five += 1
        }
        if(bills[i] === 10) {
            if(five > 0) {
                // 十元则减少五元增加十元
                five -= 1
                ten += 1
            } else {
                return false
            }
        }
        if(bills[i] === 20) {
            if(ten > 0 && five > 0) {
                // 贪心：优先使用面值大的零钱
                // 二十元则减去 10 + 5 元
                five -= 1
                ten -= 1
            } else if(five > 2) {
                // 减去三张五元
                five -= 3
            } else {
                return false
            }
        }
    }
    return true
};
```
