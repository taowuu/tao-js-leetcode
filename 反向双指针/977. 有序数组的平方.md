/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortedSquares = function(nums) {
    // 非递减则只能左边值能和右边一较高下
    // 使用反向双指针
    // 新建数组存放答案，从大到小填充
    const res = []
    let left = 0,
        right = nums.length - 1,
        end = right
    while(left <= right) {
        let leftSqr = nums[left] * nums[left],
            rightSqr = nums[right] * nums[right]
        if(leftSqr > rightSqr) {
            res[end] = leftSqr
            left += 1
        } else {
            res[end] = rightSqr
            right -= 1
        }
        end -= 1
    }
    return res
};
