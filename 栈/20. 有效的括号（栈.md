/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function(s) {
    // 括号匹配想到栈
    const stack = []
    // 缓存括号匹配情况
    const obj = {
        '(': ')',
        '[': ']',
        '{': '}',
    }
    for(let val of s) {
        if(val in obj) {
            // 左括号入栈
            stack.push(val)
        } else {
            // 右括号则判断和左括号是否匹配
            let top = stack.pop()
            if(obj[top] !== val) return false
        }
    }
    // 只有左括号则不匹配
    return stack.length === 0
};
