/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var deleteDuplicates = function(head) {
    let p = head
    // 保证下个节点存在
    while(p && p.next) {
        if(p.val === p.next.val) {
            // 连续跳过下个相同节点
            p.next = p.next.next
        } else {
            p = p.next
        }
    }
    return head
};
