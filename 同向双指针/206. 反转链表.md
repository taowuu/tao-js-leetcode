/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    // 同向双指针
    let right = head,
        left = null
    while(right) {
        // 临时保存右节点下一个
        const temp = right.next
        // 1, 右边节点往左指
        right.next = left
        // 2, 左节点前进
        left = right
        // 3, 右节点前进
        right = temp
    }
    return left
};
